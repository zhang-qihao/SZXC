    methods: {
        handleScroll(e) {
            if (e.target.scrollTop >= 0 && e.target.scrollTop < this.$refs.first.offsetHeight / 3) {
                this.selectTab = 0
            }
            if (e.target.scrollTop >= this.$refs.first.offsetHeight / 3 && e.target.scrollTop < (this.$refs.first.offsetHeight + this.$refs.second.offsetHeight/3)) {
                this.selectTab = 1
            }
            if (e.target.scrollTop >= (this.$refs.second.offsetHeight/3 + this.$refs.first.offsetHeight)) {
                this.selectTab = 2
            }
        },

                case 1:
                    that.height = that.$refs.first.offsetHeight;
                    that.selectTab = 1
                    break;
                case 2:
                    that.height = that.$refs.first.offsetHeight + that.$refs.second.offsetHeight;
                    that.selectTab = 2
                    break;
                default:
                    break;
            }
            that.$refs.orderview.scrollTop = that.height;
        },
